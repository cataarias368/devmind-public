export async function compressContext(
  messages: AgentMessage[], 
  llmProvider: LLMProvider
): Promise<AgentMessage[]> {
  // 1. Mantener siempre los últimos 10 mensajes sin comprimir
  const recentMessages = messages.slice(-10);
  
  // 2. Identificar mensajes antiguos
  const oldMessages = messages.slice(0, messages.length - 10);
  
  // 3. Filtrar mensajes importantes
  const importantMessages = oldMessages.filter(msg => {
    const content = msg.content.toLowerCase();
    return msg.role === 'system' || 
           content.includes('error') || 
           content.includes('fail') ||
           content.includes('✅') ||
           content.includes('🔒');
  });

  // 4. Generar resumen de mensajes importantes
  if (importantMessages.length === 0) return [...recentMessages];
  
  try {
    const summaryPrompt = `Resumí estos mensajes brevemente manteniendo objetivos, decisiones y errores:
    ${importantMessages.map(m => `${m.role}: ${m.content}`).join('\n')}`;
    
    // NOTA: En un entorno real, esto llama a llmProvider.call()
    // Para este snippet, simulamos la respuesta para que sea funcional
    const simulatedSummary = "[RESUMEN] Contexto inicial discutido sobre la arquitectura del proyecto.";

    return [
      { role: 'system', content: simulatedSummary, timestamp: Date.now() },
      ...recentMessages
    ];
  } catch (error) {
    console.error("Error comprimiendo contexto:", error);
    return [...recentMessages];
  }
}

import { AgentMessage } from './types';

export class SlidingWindow {
  private window: Map<string, AgentMessage> = new Map();
  private maxSize: number = 100;

  addMessage(message: AgentMessage, priority: number = 1): void {
    const id = `${Date.now()}-${Math.random()}`;
    message.priority = priority;
    message.timestamp = Date.now();
    this.window.set(id, message);

    if (this.window.size > this.maxSize) {
      this.evictLowestPriority();
    }
  }

  getActiveWindow(): AgentMessage[] {
    return Array.from(this.window.values()).sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
  }

  evictLowestPriority(): void {
    let lowestId: string | null = null;
    let lowestScore = Infinity;

    for (const [id, msg] of this.window) {
      const score = msg.priority || 0;
      if (score < lowestScore) {
        lowestScore = score;
        lowestId = id;
      }
    }

    if (lowestId) {
      this.window.delete(lowestId);
    }
  }
}

import { ToolDefinition } from './types';

export class ToolPrioritizer {
  private typeKeywords = {
    'read': ['leer', 'ver', 'get', 'show', 'list', 'file', 'path'],
    'write': ['escribir', 'crear', 'guardar', 'agregar', 'crear'],
    'search': ['buscar', 'find', 'grep', 'search'],
    'execute': ['ejecutar', 'run', 'comando', 'test', 'build'],
    'image': ['imagen', 'diagrama', 'generar', 'gráfico']
  };

  analyzeTask(task: string): string {
    const lower = task.toLowerCase();
    if (lower.includes('imagen') || lower.includes('gráfico')) return 'image';
    if (lower.includes('ejecutar') || lower.includes('test') || lower.includes('run')) return 'execute';
    if (lower.includes('buscar') || lower.includes('encuentra')) return 'search';
    return 'general';
  }

  prioritizeTools(tools: ToolDefinition[], task: string): ToolDefinition[] {
    const type = this.analyzeTask(task);
    const keywords = this.typeKeywords[type] || [];

    return tools
      .map(tool => {
        let score = 0;
        
        // Relevancia por tipo
        const toolLower = tool.name.toLowerCase();
        const descLower = tool.description.toLowerCase();
        
        keywords.forEach(kw => {
          if (toolLower.includes(kw) || descLower.includes(kw)) score += 5;
        });

        return { ...tool, score };
      })
      .sort((a, b) => b.score - a.score)
      .map(({ score, ...rest }) => rest);
  }
}

function cosineSimilarity(vecA: number[], vecB: number[]): number {
  if (vecA.length !== vecB.length) return 0;
  let dotProduct = 0;
  let magA = 0;
  let magB = 0;
  for (let i = 0; i < vecA.length; i++) {
    dotProduct += vecA[i] * vecB[i];
    magA += vecA[i] * vecA[i];
    magB += vecB[i] * vecB[i];
  }
  return magA && magB ? dotProduct / (Math.sqrt(magA) * Math.sqrt(magB)) : 0;
}

export class SemanticCache {
  private cache: Map<string, { content: string; timestamp: number; vector: number[] }> = new Map();
  private threshold = 0.85;
  private expiryMs = 60 * 60 * 1000; // 1 hora

  async get(query: string): Promise<string | null> {
    const vector = getEmbedding(query);
    
    for (const [key, data] of this.cache) {
      const now = Date.now();
      if (now - data.timestamp > this.expiryMs) {
        this.cache.delete(key);
        continue;
      }
      if (cosineSimilarity(vector, data.vector) > this.threshold) {
        return data.content;
      }
    }
    return null;
  }

  async set(query: string, result: string): Promise<void> {
    this.cache.set(query, {
      content: result,
      timestamp: Date.now(),
      vector: getEmbedding(query)
    });
  }
}

import { MemoryEntry } from './types';

export class ShortTermMemory {
  private memories: MemoryEntry[] = [];
  private maxSize = 20;

  add(action: string, result: string, importance: number): void {
    this.memories.push({
      action,
      result,
      timestamp: Date.now(),
      importance
    });

    if (this.memories.length > this.maxSize) {
      this.evictLeastImportant();
    }
  }

  getRecent(limit: number): MemoryEntry[] {
    return this.memories.slice(-limit);
  }

  getContext(): string {
    return this.memories
      .slice(-5)
      .map(m => `Acción previa: ${m.action} -> Resultado: ${m.result.slice(0, 50)}...`)
      .join('\n');
  }

  private evictLeastImportant(): void {
    this.memories.sort((a, b) => b.importance - a.importance);
    this.memories.shift();
  }
}

import { Episode } from './types';

export class EpisodicMemory {
  private episodes: Episode[] = [];
  private maxEpisodes = 100;

  add(task: string, plan: string, result: string, success: boolean): void {
    this.episodes.push({
      task,
      plan,
      result,
      success,
      timestamp: Date.now()
    });

    if (this.episodes.length > this.maxEpisodes) {
      this.episodes.shift();
    }
  }

  findSimilar(task: string): Episode[] {
    const taskWords = task.toLowerCase().split(' ');
    const matches: Episode[] = [];

    for (const ep of this.episodes) {
      const epWords = ep.task.toLowerCase().split(' ');
      let intersection = 0;
      for (const w of taskWords) {
        if (epWords.includes(w)) intersection++;
      }

      const union = new Set([...taskWords, ...epWords]).size;
      const similarity = intersection / union;

      if (similarity > 0.3) { // Umbral bajo para "similar"
        matches.push(ep);
      }
    }

    return matches.slice(0, 3); // Top 3 similares
  }
}

import { TaskResult } from './types';

export class DynamicPlanning {
  private plan: string[] = [];
  private history: { step: string; result: TaskResult }[] = [];

  createPlan(task: string): string[] {
    this.plan = [
      "Analizar requerimientos",
      "Explorar archivos",
      "Implementar cambios",
      "Verificar"
    ];
    return this.plan;
  }

  adapt(step: string, result: TaskResult): string[] {
    this.history.push({ step, result });

    if (!result.success) {
      // Replanificar si falla
      this.plan = [
        "Diagnosticar error",
        `Intentar enfoque alternativo para: ${step}`,
        ...this.plan.slice(this.plan.indexOf(step) + 1)
      ];
      return this.plan;
    }

    return this.plan;
  }

  replan(): string[] {
    const lastStep = this.history[this.history.length - 1]?.step;
    return [
      "Analizar error crítico",
      `Corregir fallo en: ${lastStep}`,
      "Continuar con el resto del plan"
    ];
  }
}

    if (batch.length >= this.maxBatchSize) {

    const batch = this.batches.get(operation) || [];
    this.batches.set(operation, []);

    return batch.map(params => ({ status: 'processed', params }));

export class SelfEvaluator {
  private metrics: Metric[] = [];

  async record(task: string, result: TaskResult): Promise<void> {
    const metric: Metric = {
      timestamp: Date.now(),
      success: result.success,
      steps: result.steps,
      tokens: 0, // Se obtiene del LLM response real
      time: 0      // Se obtiene del medidor de tiempo real
    };

    await this.save();

  async evaluate(): Promise<{ score: string; details: any }> {
    if (this.metrics.length === 0) return { score: 'no-data', details: {} };

    const total = this.metrics.length;
    const successRate = this.metrics.filter(m => m.success).length / total;
    
    let score = 'needs-improvement';
    if (successRate > 0.9 && this.getAvgSteps() < 10) score = 'excellent';
    else if (successRate > 0.7 && this.getAvgSteps() < 20) score = 'good';

    return { score, details: { successRate, avgSteps: this.getAvgSteps() } };
  }

  private getAvgSteps(): number {
    return this.metrics.reduce((acc, m) => acc + m.steps, 0) / this.metrics.length;
  }

  private async save(): Promise<void> {
    const dir = path.join(process.cwd(), '.devmind');
    await fs.mkdir(dir, { recursive: true });
    await fs.writeFile(path.join(dir, 'metrics.json'), JSON.stringify(this.metrics, null, 2));
  }
}

import * as os from 'os';

export class HealthCheck {
  check(): { status: 'healthy' | 'warning' | 'critical'; issues: string[] } {
    const issues: string[] = [];

    // 1. API Key
    if (!process.env.GLM_API_KEY) issues.push("Falta GLM_API_KEY");

    // 2. Disco (> 10% libre = Warning, < 10% = Critical)
    const freeMem = os.freemem();
    const totalMem = os.totalmem();
    const freeMemPercent = (freeMem / totalMem) * 100;

    if (freeMemPercent < 10) issues.push("Espacio en disco crítico");
    else if (freeMemPercent < 20) issues.push("Espacio en disco bajo (Warning)");

    // 3. Memoria (< 90% usado = Warning)
    const usedMemPercent = ((totalMem - freeMem) / totalMem) * 100;
    if (usedMemPercent > 90) issues.push("Uso de memoria alto");

    if (issues.length > 0 && issues.some(i => i.includes("crític"))) {
      return { status: 'critical', issues };
    }
    if (issues.length > 0) {
      return { status: 'warning', issues };
    }
    return { status: 'healthy', issues: [] };
  }
}

  async addAd(ad: Partial<AdBanner>): Promise<void> {
    const newAd = {
      ...ad,
      impressions: 0,
      clicks: 0,
      conversionRate: 0,
      isActive: true
    };
    this.ads.push(newAd);
    await this.saveAds();
  }

  async removeAd(id: string): Promise<void> {
    this.ads = this.ads.filter(a => a.id !== id);
    this.ads = this.ads.filter(a => a.id === id && !a.isActive); // Mantiene los inactivos
    this.saveAds();
    await this.saveClosedAds(); // Mantiene lista de cerrados
  }

  async updateConfig(updates: Partial<AdConfig>): Promise<void> {
    Object.assign(this.config, updates);
    const configPath = path.join(process.cwd(), '.devmind', 'ad-config.json');
    const dir = path.dirname(configPath);
    await fs.mkdir(dir, { recursive: true });
    await fs.writeFile(configPath, JSON.stringify(this.config, null, 2), 'utf-8');
  }

  async updateAd(adId: string, updates: Partial<AdBanner>): Promise<void> {
    const ad = this.ads.find(a => a.id === adId);
    if (ad) {
      Object.assign(ad, updates);
      this.ads = this.ads.map(a => a.id === adId ? { ...a, ...updates } : a);
      await this.saveAds();
    }
  }

  async closeAd(adId: string): void {
    this.closedAds.add(adId);
    this.ads = this.ads.map(a => a.id !== adId && !a.isActive);
    this.saveAds();
  }

  // --- CONFIGURACIÓN ---
  
  getConfig(): AdConfig {
    return this.config;
  }
