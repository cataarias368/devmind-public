    38вҶ’## рҹҡҖ ВҝQuГ© es DevMind?
    39вҶ’
    40вҶ’DevMind es un **agente de software autГіnomo** que puede planificar, ejecutar, probar, documentar y desplegar cГіdigo sin intervenciГіn humana. Es como tener un equipo de desarrolladores junior 24/7, supervisado por un arquitecto senior.
    41вҶ’
    42вҶ’### вңЁ CaracterГӯsticas Principales
    43вҶ’
    44вҶ’| CaracterГӯstica | DescripciГіn |
    45вҶ’|----------------|-------------|
    46вҶ’| рҹӨ– **Agente AutГіnomo** | Planifica y ejecuta tareas de desarrollo sin supervisiГіn |
    47вҶ’| рҹ§  **Multi-Agente** | 5 agentes especializados trabajando en paralelo |
    48вҶ’| рҹҺЁ **GeneraciГіn de ImГЎgenes** | ГҚconos, diagramas, mockups con CogView |
    49вҶ’| рҹ“ҡ **DocumentaciГіn** | Genera MD/HTML/PDF automГЎticamente |
    50вҶ’| рҹ§Ә **Tests** | Unitarios, integraciГіn y E2E con Playwright |
    51вҶ’| рҹ“Ҡ **15 MГіdulos de Rendimiento** | Cache semГЎntico, planificaciГіn dinГЎmica, etc. |
    52вҶ’| рҹ”Ң **Plugins** | Extensible con herramientas personalizadas |
    53вҶ’| рҹҢҗ **Dashboard Web** | Interfaz grГЎfica para controlar el agente |
    54вҶ’| рҹ’¬ **Bots** | Slack, Discord y Telegram |
    55вҶ’| рҹҗҷ **GitHub** | Issues, PRs y bГәsqueda de cГіdigo |
    56вҶ’| рҹ“ў **MonetizaciГіn** | Anuncios no intrusivos y suscripciones |

   134вҶ’## рҹ”Ң Plugins
   135вҶ’
   136вҶ’DevMind es extensible mediante plugins. Los plugins pueden:
   137вҶ’
   138вҶ’- Agregar nuevas herramientas
   139вҶ’- Modificar el comportamiento del agente
   140вҶ’- Integrarse con sistemas externos
   141вҶ’
   142вҶ’### Ejemplo de Plugin
   143вҶ’
   144вҶ’```typescript
   145вҶ’// plugins/my-plugin.ts
   146вҶ’export default {
   147вҶ’  name: 'my-plugin',
   148вҶ’  version: '1.0.0',
   149вҶ’  description: 'Mi plugin personalizado',
   150вҶ’  
   151вҶ’  // Herramientas que agrega
   152вҶ’  tools: [
   153вҶ’    {
   154вҶ’      name: 'deploy_to_aws',
   155вҶ’      description: 'Despliega el proyecto a AWS',
   156вҶ’      parameters: {
   157вҶ’        type: 'object',
   158вҶ’        properties: {
   159вҶ’          region: { type: 'string' },
   160вҶ’          env: { type: 'string' }
   161вҶ’        }
   162вҶ’      }
   163вҶ’    }
   164вҶ’  ],
   165вҶ’  
   166вҶ’  // Permisos que necesita
   167вҶ’  permissions: [
   168вҶ’    'read:files',
   169вҶ’    'write:temp',
   170вҶ’    'execute:aws-cli'
   171вҶ’  ],
   172вҶ’  
   173вҶ’  // CГіdigo que se ejecuta al cargar
   174вҶ’  onLoad: (core) => {
   175вҶ’    console.log('Plugin cargado correctamente');
   176вҶ’  }
   177вҶ’};
   178вҶ’```
   179вҶ’
   180вҶ’### Sistema de Permisos
   181вҶ’
   182вҶ’Los plugins deben declarar los permisos que necesitan:
   183вҶ’
   184вҶ’| Permiso | DescripciГіn |
   185вҶ’|---------|-------------|
   186вҶ’| `read:files` | Leer archivos del workspace |
   187вҶ’| `write:files` | Escribir archivos del workspace |
   188вҶ’| `write:temp` | Escribir en archivos temporales |
   189вҶ’| `execute:command` | Ejecutar comandos del sistema |
   190вҶ’| `execute:npm` | Ejecutar comandos npm |
   191вҶ’| `execute:git` | Ejecutar comandos git |
   192вҶ’| `network:http` | Hacer peticiones HTTP |
   193вҶ’| `network:https` | Hacer peticiones HTTPS |
   194вҶ’| `storage:memory` | Leer/escribir memoria persistente |

   581вҶ’## рҹ”Ң 4. SISTEMA DE PLUGINS CON PERMISOS
   582вҶ’
   583вҶ’### Archivo: `src/plugin-system.ts` (VersiГіn completa con permisos)
   584вҶ’
   585вҶ’```typescript
   586вҶ’// src/plugin-system.ts
   587вҶ’import * as fs from 'fs/promises';
   588вҶ’import * as path from 'path';
   589вҶ’import { AgentCore } from './types';
   590вҶ’
   591вҶ’export type Permission =
   592вҶ’  | 'read:files'
   593вҶ’  | 'write:files'
   594вҶ’  | 'write:temp'
   595вҶ’  | 'execute:command'
   596вҶ’  | 'execute:npm'
   597вҶ’  | 'execute:git'
   598вҶ’  | 'network:http'
   599вҶ’  | 'network:https'
   600вҶ’  | 'storage:memory'
   601вҶ’  | 'storage:persistent';
   602вҶ’
   603вҶ’export interface Plugin {
   604вҶ’  name: string;
   605вҶ’  version: string;
   606вҶ’  description?: string;
   607вҶ’  permissions: Permission[];
   608вҶ’  tools?: PluginTool[];
   609вҶ’  onLoad?: (core: AgentCore) => Promise<void> | void;
   610вҶ’  onUnload?: () => Promise<void> | void;
   611вҶ’}
   612вҶ’
   613вҶ’export interface PluginTool {
   614вҶ’  name: string;
   615вҶ’  description: string;
   616вҶ’  parameters: {
   617вҶ’    type: 'object';
   618вҶ’    properties: Record<string, any>;
   619вҶ’    required: string[];
   620вҶ’  };
   621вҶ’  handler: (args: any, core: AgentCore, permissions: Permission[]) => Promise<string>;
   622вҶ’}
   623вҶ’
   624вҶ’export interface PluginManifest {
   625вҶ’  name: string;
   626вҶ’  version: string;
   627вҶ’  description?: string;
   628вҶ’  permissions: Permission[];
   629вҶ’  main: string;
   630вҶ’}
   631вҶ’
   632вҶ’export class PluginManager {
   633вҶ’  private plugins: Map<string, Plugin> = new Map();
   634вҶ’  private core: AgentCore;
   635вҶ’  private loaded: Set<string> = new Set();
   636вҶ’  private manifestPath: string;
   637вҶ’
   638вҶ’  constructor(core: AgentCore) {
   639вҶ’    this.core = core;
   640вҶ’    this.manifestPath = path.join(process.cwd(), '.devmind', 'plugins', 'manifest.json');
   641вҶ’  }
   642вҶ’
   643вҶ’  // ============================================================
   644вҶ’  // CARGA DE PLUGINS
   645вҶ’  // ============================================================
   646вҶ’
   647вҶ’  async loadPlugin(plugin: Plugin): Promise<void> {
   648вҶ’    if (this.loaded.has(plugin.name)) {
   649вҶ’      console.warn(`[Plugin] ${plugin.name} ya estГЎ cargado`);
   650вҶ’      return;
   651вҶ’    }
   652вҶ’
   653вҶ’    // Validar permisos
   654вҶ’    const hasAllPermissions = await this.validatePermissions(plugin.permissions);
   655вҶ’    if (!hasAllPermissions) {
   656вҶ’      console.warn(`[Plugin] ${plugin.name} no tiene permisos suficientes`);
   657вҶ’      return;
   658вҶ’    }
   659вҶ’
   660вҶ’    try {
   661вҶ’      if (plugin.onLoad) {
   662вҶ’        await plugin.onLoad(this.core);
   663вҶ’      }
   664вҶ’      this.plugins.set(plugin.name, plugin);
   665вҶ’      this.loaded.add(plugin.name);
   666вҶ’      console.log(`[Plugin] вң… Cargado: ${plugin.name} v${plugin.version}`);
   667вҶ’    } catch (err: any) {
   668вҶ’      console.error(`[Plugin] вқҢ Error cargando ${plugin.name}:`, err.message);
   669вҶ’      throw err;
   670вҶ’    }
   671вҶ’  }
   672вҶ’
   673вҶ’  async loadPluginFromFile(filePath: string): Promise<void> {
   674вҶ’    const fullPath = path.resolve(filePath);
   675вҶ’    try {
   676вҶ’      const module = await import(fullPath);
   677вҶ’      const plugin = module.default || module;
   678вҶ’      if (!plugin.name || !plugin.version) {
   679вҶ’        throw new Error('El plugin debe exportar { name, version }');
   680вҶ’      }
   681вҶ’      await this.loadPlugin(plugin);
   682вҶ’    } catch (err: any) {
   683вҶ’      console.error(`[Plugin] вқҢ Error cargando desde ${filePath}:`, err.message);
   684вҶ’      throw err;
   685вҶ’    }
   686вҶ’  }
   687вҶ’
   688вҶ’  async loadPluginsFromDirectory(dir: string): Promise<number> {
   689вҶ’    let count = 0;
   690вҶ’    try {
   691вҶ’      const files = await fs.readdir(dir);
   692вҶ’      for (const file of files) {
   693вҶ’        if (file.endsWith('.js') || file.endsWith('.ts')) {
   694вҶ’          await this.loadPluginFromFile(path.join(dir, file));
   695вҶ’          count++;
   696вҶ’        }
   697вҶ’      }
   698вҶ’    } catch (err) {
   699вҶ’      console.warn(`[Plugin] No se pudo leer el directorio ${dir}:`, err);
   700вҶ’    }
   701вҶ’    return count;
   702вҶ’  }

   705вҶ’  // VALIDACIГ“N DE PERMISOS
   706вҶ’  // ============================================================
   707вҶ’
   708вҶ’  private async validatePermissions(permissions: Permission[]): Promise<boolean> {
   709вҶ’    const userApproved = await this.getUserApprovedPermissions();
   710вҶ’
   711вҶ’    for (const perm of permissions) {
   712вҶ’      if (!userApproved.includes(perm)) {
   713вҶ’        const approval = await this.requestPermission(perm);
   714вҶ’        if (!approval) {
   715вҶ’          console.warn(`[Plugin] Permiso denegado: ${perm}`);
   716вҶ’          return false;
   717вҶ’        }
   718вҶ’        userApproved.push(perm);
   719вҶ’        await this.saveApprovedPermissions(userApproved);
   720вҶ’      }
   721вҶ’    }
   722вҶ’
   723вҶ’    return true;
   724вҶ’  }
   725вҶ’
   726вҶ’  private async getUserApprovedPermissions(): Promise<Permission[]> {
   727вҶ’    try {
   728вҶ’      const data = await fs.readFile(this.manifestPath, 'utf-8');
   729вҶ’      const manifest = JSON.parse(data);
   730вҶ’      return manifest.approvedPermissions || [];
   731вҶ’    } catch {
   732вҶ’      return [];
   733вҶ’    }
   734вҶ’  }
   735вҶ’
   736вҶ’  private async saveApprovedPermissions(permissions: Permission[]): Promise<void> {
   737вҶ’    const dir = path.dirname(this.manifestPath);
   738вҶ’    await fs.mkdir(dir, { recursive: true });
   739вҶ’    await fs.writeFile(
   740вҶ’      this.manifestPath,
   741вҶ’      JSON.stringify({ approvedPermissions: permissions }, null, 2),
   742вҶ’      'utf-8'
   743вҶ’    );
   744вҶ’  }
   745вҶ’
   746вҶ’  private async requestPermission(permission: Permission): Promise<boolean> {
   747вҶ’    // En CLI, preguntar al usuario
   748вҶ’    console.log(`\nрҹ”җ El plugin solicita el permiso: ${permission}`);
   749вҶ’    console.log(`   DescripciГіn: ${this.getPermissionDescription(permission)}`);
   750вҶ’    console.log(`   ВҝPermitir? (y/N)`);
   751вҶ’
   752вҶ’    // Simple implementaciГіn para CLI
   753вҶ’    return new Promise((resolve) => {
   754вҶ’      process.stdin.once('data', (data) => {
   755вҶ’        const answer = data.toString().trim().toLowerCase();
   756вҶ’        resolve(answer === 'y' || answer === 'yes');
   757вҶ’      });
   758вҶ’    });
   759вҶ’  }

   768вҶ’      'execute:git': 'Ejecutar comandos git',
   769вҶ’      'network:http': 'Hacer peticiones HTTP',

   772вҶ’      'storage:persistent': 'Leer/escribir memoria persistente',

   774вҶ’    return descriptions[permission] || 'Permiso desconocido';
   775вҶ’  }

   778вҶ’  // EJECUCIГ“N DE HERRAMIENTAS CON PERMISOS
   779вҶ’  // ============================================================
   780вҶ’
   781вҶ’  async executeTool(
   782вҶ’    pluginName: string,
   783вҶ’    toolName: string,
   784вҶ’    args: any
   785вҶ’  ): Promise<string> {
   786вҶ’    const plugin = this.plugins.get(pluginName);
   787вҶ’    if (!plugin) {
   788вҶ’      throw new Error(`Plugin ${pluginName} no cargado`);
   789вҶ’    }
   790вҶ’
   791вҶ’    const tool = plugin.tools?.find(t => t.name === toolName);
   792вҶ’    if (!tool) {
   793вҶ’      throw new Error(`Herramienta ${toolName} no encontrada en ${pluginName}`);
   794вҶ’    }
   795вҶ’
   796вҶ’    return await tool.handler(args, this.core, plugin.permissions);
   797вҶ’  }
