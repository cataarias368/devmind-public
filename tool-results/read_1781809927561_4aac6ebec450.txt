   169→    const type = this.analyzeTask(task);
   170→    const keywords = this.typeKeywords[type] || [];
   171→
   172→    return tools
   173→      .map(tool => {
   174→        let score = 0;
   175→        
   176→        // Relevancia por tipo

   178→        const descLower = tool.description.toLowerCase();

   180→        keywords.forEach(kw => {
   181→          if (toolLower.includes(kw) || descLower.includes(kw)) score += 5;
   182→        });
   183→
   184→        return { ...tool, score };
   185→      })
   186→      .sort((a, b) => b.score - a.score)
   187→      .map(({ score, ...rest }) => rest);

   201→function cosineSimilarity(vecA: number[], vecB: number[]): number {
   202→  if (vecA.length !== vecB.length) return 0;
   203→  let dotProduct = 0;
   204→  let magA = 0;
   205→  let magB = 0;
   206→  for (let i = 0; i < vecA.length; i++) {
   207→    dotProduct += vecA[i] * vecB[i];
   208→    magA += vecA[i] * vecA[i];
   209→    magB += vecB[i] * vecB[i];
   210→  }
   211→  return magA && magB ? dotProduct / (Math.sqrt(magA) * Math.sqrt(magB)) : 0;
   212→}

   216→  private threshold = 0.85;

   219→  async get(query: string): Promise<string | null> {
   220→    const vector = getEmbedding(query);

   222→    for (const [key, data] of this.cache) {

   225→        this.cache.delete(key);

   229→        return data.content;

   231→    }

   233→  }

   306→  findSimilar(task: string): Episode[] {

   309→
   310→    for (const ep of this.episodes) {
   311→      const epWords = ep.task.toLowerCase().split(' ');
   312→      let intersection = 0;
   313→      for (const w of taskWords) {
   314→        if (epWords.includes(w)) intersection++;
   315→      }
   316→
   317→      const union = new Set([...taskWords, ...epWords]).size;
   318→      const similarity = intersection / union;
   319→
   320→      if (similarity > 0.3) { // Umbral bajo para "similar"
   321→        matches.push(ep);
   322→      }
   323→    }
